# define user-configurable options to build the smiletool and /or shapes executables;

    add_subdirectory(wizard)
endif()
if (BUILD_SMILE_TOOL)
    add_subdirectory(smiletool)
endif()
if (BUILD_SMILE_SHAPES)
    add_subdirectory(shapes)
endif()
